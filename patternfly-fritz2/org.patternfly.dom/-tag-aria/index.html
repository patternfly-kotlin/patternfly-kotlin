TagAria

class TagAria<E : Element>(tag: Tag<E>) : ElementAria
Content copied to clipboard

Class that simplifies the handling of ARIA attributes on Tags in a way that you don't have to specify the aria- prefix.

Samples

import dev.fritz2.dom.html.render
import kotlinx.browser.document
import org.patternfly.dom.aria
fun main() { 
   //sampleStart 
   render {
    val tag = div {
        +"Some text"
    }
    tag.aria["label"] = "More info"
    val yes = "label" in tag.aria
    val moreInfo = tag.aria["label"]
} 
   //sampleEnd
}
