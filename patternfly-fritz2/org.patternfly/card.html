Creates a Card component which is part of an CardView.

import dev.fritz2.dom.html.render
import org.patternfly.Align.RIGHT
import org.patternfly.ItemStore
import org.patternfly.card
import org.patternfly.cardAction
import org.patternfly.cardBody
import org.patternfly.cardCheckbox
import org.patternfly.cardHeader
import org.patternfly.cardTitle
import org.patternfly.cardView
import org.patternfly.dropdown
import org.patternfly.item
import org.patternfly.items
import org.patternfly.kebabToggle
fun main() { 
   //sampleStart 
   render {
    data class Demo(val id: String, val name: String)

    val store = ItemStore<Demo> { it.id }
    cardView(store) {
        display { demo ->
            card(demo) {
                cardHeader {
                    cardTitle { +"Demo" }
                    cardAction {
                        dropdown<String>(align = RIGHT) {
                            kebabToggle()
                            items {
                                item("Edit")
                                item("Remove")
                            }
                        }
                        cardCheckbox()
                    }
                }
                cardBody(id = itemId(demo)) { +demo.name }
            }
        }
    }

    store.addAll(
        listOf(
            Demo("foo", "Foo"),
            Demo("bar", "Bar")
        )
    )
} 
   //sampleEnd
}

Creates a standalone Card component.

Samples

import dev.fritz2.dom.html.render
import org.patternfly.Align.RIGHT
import org.patternfly.card
import org.patternfly.cardAction
import org.patternfly.cardBody
import org.patternfly.cardCheckbox
import org.patternfly.cardFooter
import org.patternfly.cardHeader
import org.patternfly.cardTitle
import org.patternfly.dropdown
import org.patternfly.item
import org.patternfly.items
import org.patternfly.kebabToggle
import org.patternfly.separator
fun main() { 
   //sampleStart 
   render {
    card {
        cardHeader {
            img { src("./logo.svg") }
            cardAction {
                dropdown<String>(align = RIGHT) {
                    kebabToggle()
                    items {
                        item("Item 1")
                        item("Disabled Item") {
                            disabled = true
                        }
                        separator()
                        item("Separated Item")
                    }
                }
                cardCheckbox()
            }
        }
        cardTitle { +"Title" }
        cardBody { +"Body" }
        cardFooter { +"Footer" }
    }
} 
   //sampleEnd
}
